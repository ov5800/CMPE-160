library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity hw12_p2 is
	Port ( 
		signal ADDR : in STD_LOGIC_VECTOR(1 DOWNTO 0);
		signal Z : out STD_LOGIC_VECTOR(2 DOWNTO 0));
end hw12_p2;

architecture behavioral of hw12_p2 is
	begin
	-- outside all processes
	process ( ADDR ) begin
		case ADDR is
			when "01" | "10" => Z <= "101" after 20 ns;
			when "11" => Z <= "111" after 20 ns;
			when others => Z <= "000" after 50 ns;
		end case;
	end process;
end behavioral;
